version: '3.9'
services:
  rabbitmq:
    build:
      context: ./Server/rabbitmq
      dockerfile: rabbitmq.dockerfile
    ports:
      - 15672:15672
  
  server:
    build:
      context: ./Server
      dockerfile: server.dockerfile
    restart: on-failure
    depends_on:
      - rabbitmq
    links: 
      - rabbitmq    
    networks:
      - testing_net
    environment:
      - PYTHONUNBUFFERED=1

  joiner:
    build:
      context: ./Server/Joiner
      dockerfile: joiner.dockerfile
    restart: on-failure
    depends_on:
      - rabbitmq
    links: 
      - rabbitmq
    environment:
      - PYTHONUNBUFFERED=1
    
  client:
    build:
      context: ./Client/Source
      dockerfile: client.dockerfile
    entrypoint: python3 /main.py
    environment:
      - PYTHONUNBUFFERED=1
    networks:
      - testing_net
    depends_on:
      - server
    volumes:
      - ./Client/config.ini:/config.ini
      - ./Client/montreal:/montreal
      - ./Client/toronto:/toronto
      - ./Client/washington:/washington
