version: '3.9'
services:
  rabbitmq:
    build:
      context: ./Server/rabbitmq
      dockerfile: rabbitmq.dockerfile
    ports:
      - 15672:15672
    networks:
      - testing_net
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:15672"]
      interval: 10s
      timeout: 5s
      retries: 10
    environment:
      - RABBITMQ_LOGS=-
  
  server:
    container_name: server
    image: server:latest
    restart: on-failure
    depends_on:
      rabbitmq:
        condition: service_healthy
    links: 
      - rabbitmq
    networks:
      - testing_net
    environment:
      - PYTHONUNBUFFERED=1
    volumes:
      - ./Server/Server/config.ini:/config.ini
